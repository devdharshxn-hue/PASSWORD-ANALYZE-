import re
import random
import string

password = input("Enter your password: ")

score = 0
suggestions = []

# Check password length
if len(password) >= 12:
    score += 2
elif len(password) >= 8:
    score += 1
else:
    suggestions.append("Use at least 12 characters.")

# Check uppercase letters
if re.search(r"[A-Z]", password):
    score += 1
else:
    suggestions.append("Add uppercase letters.")

# Check lowercase letters
if re.search(r"[a-z]", password):
    score += 1
else:
    suggestions.append("Add lowercase letters.")

# Check numbers
if re.search(r"\d", password):
    score += 1
else:
    suggestions.append("Add numbers.")

# Check special characters
if re.search(r"[!@#$%^&*(),.?\":{}|<>]", password):
    score += 1
else:
    suggestions.append("Add special characters.")

# Determine password strength
if score <= 2:
    strength = "Weak"
elif score <= 5:
    strength = "Moderate"
else:
    strength = "Strong"

print("\nPassword Strength:", strength)

# Generate strong password suggestion
characters = string.ascii_letters + string.digits + "!@#$%^&*"
strong_password = ''.join(random.choice(characters) for _ in range(12))

if suggestions:
    print("\nSuggestions to improve your password:")
    for suggestion in suggestions:
        print("-", suggestion)

    print("\nSuggested Strong Password:", strong_password)

# Save result to a report file
with open("Password_Report.txt", "w") as file:
    file.write("PASSWORD STRENGTH ANALYZER REPORT\n")
    file.write("---------------------------------\n")
    file.write(f"Entered Password: {password}\n")
    file.write(f"Password Strength: {strength}\n\n")
    
    if suggestions:
        file.write("Suggestions:\n")
        for suggestion in suggestions:
            file.write(f"- {suggestion}\n")
        
        file.write(f"\nSuggested Strong Password: {strong_password}\n")

print("\nReport saved as 'Password_Report.txt'")
